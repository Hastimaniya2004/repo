git remote -v
# Add a remote manually (if not already added)
git remote add origin https://github.com/username/repository.git

git fetch origin

git pull origin main
# Make local changes
echo "New change" >> app.txt

=======================================================
Undoing Changes & Recovery
# Undo unstaged changes (restore file to last commit)
git checkout -- app.txt
# or: git restore app.txt
# Unstage a file (keep changes in working directory)
git reset app.txt
# Amend last commit (modify message or add changes)
git commit --amend -m "Updated commit message"
# ---- Using git reset ----
# Soft reset (keeps changes staged)
git reset --soft HEAD~1
# Mixed reset (default: keeps changes unstaged)
git reset HEAD~1
# Hard reset (removes all changes permanently)
git reset --hard HEAD~1
# ---- Safe undo using revert ----
git revert HEAD
# ---- Recover lost commits ----
git reflog
# Restore a commit from reflog
git checkout <commit-hash>
# or reset back to it
git reset --hard <commit-hash>

# Check status

# Stage file

git diff

git commit -m "Updated app.txt"
# -------- GIT HOOKS FOR AUTOMATION --------
# Go to hooks directory
cd .git/hooks
# -------- Pre-commit hook (lint check) --------
# Create/edit pre-commit file
echo '#!/bin/sh
echo "Running lint check..."
# Example: replace with real linter
exit 0' > pre-commit
# Make it executable
chmod +x pre-commit
# -------- Pre-push hook (run tests) --------
echo '#!/bin/sh
echo "Running tests before push..."
# Example: replace with real test command
exit 0' > pre-push
chmod +x pre-push
# -------- Post-merge hook --------
echo '#!/bin/sh
echo "Post-merge actions running..."
# Example: install dependencies or build
exit 0' > post-merge
chmod +x post-merge
# -------- BACK TO PROJECT ROOT --------
cd ../..
# Now hooks will run automatically on commit/push/merge

# Delete remote branch
git push origin --delete dev
# Remove merged branches (cleanup)
git branch --merged | grep -v "\*" | xargs -n 1 git branch -d
# Clean unused remote-tracking branches
git fetch --prune
# Verify branch list
git branch
